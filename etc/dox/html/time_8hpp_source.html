275  static timespec utc2timespec(std::uint64_t value)
276  {
277  // Parses SendingTime, pattern is number representation in format yyyymmddhhmmssnnnnn
278  // where nnnnn is milliseconds/microseconds/nanoseconds value
279  // Note: only up to 5 nanosec digits can safely fit uint64_t in this format
280  constexpr static uint date_prefix_len{ 8U };
281  constexpr static uint tm_part_len{ 2U };
282  constexpr static uint ns_len{ 9U };
283 
284  auto timestamp_len = num_digits(value);
285  // @warning: assert(num_digits(value) >= 14);
286 
287  timestamp_len -= date_prefix_len;
288  auto timestamp_value = value % pow10u(timestamp_len);
289 
290  timestamp_len -= tm_part_len;
291  const uint h = timestamp_value / pow10u(timestamp_len);
292  timestamp_value %= pow10u(timestamp_len);
293 
294  timestamp_len -= tm_part_len;
295  const uint m = timestamp_value / pow10u(timestamp_len);
296  timestamp_value %= pow10u(timestamp_len);
297 
298  timestamp_len -= tm_part_len;
299  const uint s = timestamp_value / pow10u(timestamp_len);
300  timestamp_value %= pow10u(timestamp_len);
301 
302  return utc2timespec(h, m, s, timestamp_value * pow10u(ns_len - timestamp_len));
303  }
