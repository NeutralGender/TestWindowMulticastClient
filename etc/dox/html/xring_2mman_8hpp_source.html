192 u_char* xring_alloc_t<xring_producer::MP>::alloc(memsize_type blk_capacity)
193 {
194  // Calculate chunks
195  const capacity_type chunks_desired = cache_chunks_ceil(xslot_t::hdr_size() + blk_capacity);
196 
197  // Update tail memory sequence
198  // Please note that msq.fetch_add(...) returns its preceding value i. e. currently processing
199  // sequence
200  sequence_type active_msq{ 0 };
201  sequence_type desired_msq{ 0 };
202  bool is_hole_slot{ false };
203  do
204  {
205  // Currently processing
206  active_msq = msq.load(std::memory_order_relaxed);
207 
208  const capacity_type chunks_avail = capacity - memseq2ix(active_msq);
209  const capacity_type chunks_allowed{ std::min(chunks_avail, chunks_desired) };
210  desired_msq = active_msq + chunks_allowed;
211  is_hole_slot = chunks_allowed < chunks_desired;
212 
213  // Try to make changes if 'msq' remains unchanged, otherwise
214  [[ likely ]]
215  BREAK_IF(msq.compare_exchange_weak(active_msq,
216  desired_msq,
217  std::memory_order_relaxed,
218  std::memory_order_relaxed));
219  }
220  while (1);
221 
222  // Construct slot in-place
223  auto placement = memseq2mem(active_msq);
224 
225  // Header updation
226  // Wait for current header sequence becomes available
227  COMPILER_BARRIER;
228  while(not msq_hdr.compare_exchange_weak(active_msq,
229  desired_msq,
230  std::memory_order_relaxed) );
231  COMPILER_BARRIER;
232 
233  auto slot = new (placement) xslot_t(chunks_desired, active_msq);
234 
235  // Check for sufficient continuous memory
236  [[ unlikely ]]
237  if (is_hole_slot)
238  {
239  slot->treat_as_hole();
240  slot->commit();
241  return alloc(blk_capacity);
242  }
243 
244  return slot->payload<u_char>();
